def draw_normal
        @main.clear do
                #@internet_status = internet_status
                #$DATA.internet_available? ? @internet_status.switch_on : @internet_status.switch_off
                @progress_bar = progress width: 700, fraction: @prog
                @message = para strong(@msg), size: 10, stroke: white
        end #main.clear
end
