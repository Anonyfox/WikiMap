def initialize
        # set up the database file, creates one if doesn't exist yet
        @home = ENV['HOME']
        @path = @home + "/.wikimap"
        Dir.mkdir @path unless Dir.exists? @path
        @name = @path + "/wikimap.db"
        @db = Kioku.new @name
end
