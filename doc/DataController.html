def initialize
        require 'kioku'
        require_relative '../libs/wiki_client'

        @home = ENV['HOME']
        @path = @home + "/.wikimap"
        Dir.mkdir @path unless Dir.exists? @path
        Dir.mkdir @path+"/tmp" unless Dir.exists? @path+"/tmp"
        @name = @path + "/wikimap.db"
        @db = Kioku.new @name
end

def internet_available?
        require 'net/ping/http'
        # google is the most solid test target
        http = Net::Ping::HTTP.new("http://www.google.com")
        http.ping? #? true : false
end
    look_for(name) click to toggle source

    returns the requested data for the given name

    # File app/controllers/data_controller.rb, line 29
def look_for name
        answer = nil
        if internet_available?
                answer = WikiClient.get name
                @db[name] = answer
        else
                answer = @db[name]
        end
        answer || []
end

def search_matching_links_to name
        answer = nil
        if internet_available?
                answer = WikiClient.ask name
        else
                answer = @db.search name
        end
        answer || []
end
